ភាគទី 16: ជំរុញខ្លួនឯងឱ្យមានសកម្មភាព

បរិភោគកង្កែបនោះ!តាមការជម្រុញខ្លួនអ្នកទៅជាសកម្មភាពដោះស្រាយសារៈសំខាន់នៃការលើកទឹកចិត្តខ្លួនឯងនិងចាត់វិធានការសកម្មដើម្បីជំនះការពន្យារពេល។នេះជាវិធីជម្រុញខ្លួនអ្នកឱ្យមានប្រសិទ្ធភាពចំពោះសកម្មភាព:

កំណត់គោលដៅរបស់អ្នក:

ចាប់ផ្តើមដោយការកំណត់គោលដៅរបស់អ្នកយ៉ាងច្បាស់។តើអ្នកចង់សម្រេចបាននូវអ្វី?មានគោលបំណងជាក់លាក់ក្នុងចិត្តនឹងផ្តល់ឱ្យអ្នកនូវគោលបំណងច្បាស់លាស់នៃគោលបំណង។
កំណត់ពេលវេលាកំណត់:

កំណត់ពេលវេលាកំណត់ទៅគោលដៅនិងភារកិច្ចរបស់អ្នក។ការកំណត់រយៈពេលនៃពេលវេលាបង្កើតភាពបន្ទាន់និងគណនេយ្យភាពជំរុញឱ្យអ្នកធ្វើសកម្មភាព។
បញ្ឈប់ភារកិច្ចចុះក្រោម:

ចែកភារកិច្ចធំ ៗ ទៅជាជំហានតូចជាងមុនអាចគ្រប់គ្រងបាន។នេះធ្វើឱ្យគោលដៅរបស់អ្នកហាក់ដូចជាមិនសូវមានការចាប់អារម្មណ៍និងអនុញ្ញាតឱ្យអ្នកផ្តោតលើកិច្ចការមួយក្នុងពេលតែមួយ។
ផ្តល់អាទិភាពដល់ភារកិច្ច:

កំណត់ភារកិច្ចសំខាន់និងផលប៉ះពាល់ដែលមានឥទ្ធិពលបំផុតលើបញ្ជីការងារត្រូវធ្វើរបស់អ្នក។អាទិភាពធានាថាអ្នកកំពុងធ្វើការលើភារកិច្ចដែលតម្រឹមតាមគោលដៅរបស់អ្នក។
បង្កើតផែនការ:

បង្កើតផែនការសកម្មភាពលម្អិតសម្រាប់ភារកិច្ចនីមួយៗ។ការមានផែនទីបង្ហាញផ្លូវធ្វើឱ្យវាកាន់តែងាយស្រួលក្នុងការមើលអ្វីដែលត្រូវធ្វើនិងរបៀបទៅទីនោះ។
6. មើលឃើញជោគជ័យ:

ប្រើថាមពលនៃការមើលឃើញដើម្បីស្រមៃមើលថាខ្លួនអ្នកដោយជោគជ័យក្នុងការបំពេញភារកិច្ចរបស់អ្នកដោយជោគជ័យនិងសម្រេចបាននូវគោលដៅរបស់អ្នក។ការមើលឃើញអាចជម្រុញការលើកទឹកចិត្តរបស់អ្នក។
ស្វែងរកមូលហេតុរបស់អ្នក:

យល់ពីមូលហេតុដែលគោលដៅរបស់អ្នកសំខាន់សម្រាប់អ្នក។គោលដៅរបស់អ្នកកាន់តែមានអត្ថន័យគឺការលើកទឹកចិត្តកាន់តែច្រើនអ្នកនឹងធ្វើការឆ្ពោះទៅរកពួកគេ។
លុបបំបាត់ការរំខាន:

កំណត់អត្តសញ្ញាណនិងលុបបំបាត់ការរំខានដែលរារាំងដល់ផលិតភាពរបស់អ្នក។បង្កើតបរិយាកាសការងារដែលផ្តោតអារម្មណ៍។
ផ្តល់រង្វាន់ដល់ខ្លួនអ្នក:

ផ្តល់រង្វាន់ដល់ខ្លួនអ្នកសម្រាប់ការបំពេញភារកិច្ចឬការឈានដល់ចំណុចកំពូល។ការពង្រឹងវិជ្ជមានអាចបង្កើនការលើកទឹកចិត្ត។
រក្សាឱ្យមានផ្នត់គំនិតវិជ្ជមាន:
បណ្ដុះឥរិយាបថវិជ្ជមានចំពោះគោលដៅនិងភារកិច្ចរបស់អ្នក។ផ្នត់គំនិតវិជ្ជមានអាចជួយអ្នកឱ្យយកឈ្នះលើបញ្ហាប្រឈមនិងការជម្រុញឱ្យមានកម្លាំងចិត្ត។
ស្វែងរកការទទួលខុសត្រូវ:
ចែករំលែកគោលដៅរបស់អ្នកជាមួយមិត្តភក្តិសមាជិកគ្រួសារឬមិត្តរួមការងារដែលអាចទទួលខុសត្រូវបាន។ដោយដឹងថាមាននរណាម្នាក់ដឹងអំពីគោលដៅរបស់អ្នកអាចផ្តល់នូវការលើកទឹកចិត្ត។
ស្វែងរកការបំផុសគំនិត:
រកមើលប្រភពនៃការបំផុសគំនិតមិនថាវាកំពុងអានរឿងរ៉ាវជោគជ័យដោយមើលវីដេអូលើកទឹកចិត្តឬភ្ជាប់ជាមួយបុគ្គលដែលមានគំនិតដូចគ្នា។
