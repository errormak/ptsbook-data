ភាគទី 20: អ្វីដែលមនុស្សគ្រប់គ្នាចង់បាន

ការស្វែងរករបស់បុរសរបស់វីកាទ័រអេ។នៅក្នុងវគ្គនេះយើងបានបង្កើតនូវបំណងប្រាថ្នាជាសកលដែលមានព្រំប្រទល់វប្បធម៌ខាងសាច់ឈាមនិងឯកត្តជន - ការស្វែងរកអត្ថន័យ។

ការស្វែងរកសកលសម្រាប់អត្ថន័យ:
ហ្វ្រែនចូសសង្កេតឃើញថាដោយមិនគិតពីកាលៈទេសៈរបស់ពួកគេប្រជាជនចែករំលែកនូវបំណងប្រាថ្នាជាមូលដ្ឋានសម្រាប់អត្ថន័យក្នុងជីវិតរបស់ពួកគេ។ការស្វែងរកគោលបំណងនេះគឺជាលក្ខណៈកំណត់នៃអត្ថិភាពរបស់មនុស្ស។

លើសពីការសប្បាយនិងសុភមង្គល:
ខណៈពេលដែលភាពរីករាយនិងសុភមង្គលគឺច្រើនដងហើយជារឿយៗពឹងផ្អែកលើកត្តាខាងក្រៅការស្វែងរកអត្ថន័យគឺជាការស្វែងរកកាន់តែស៊ីជម្រៅដែលកាន់តែស៊ូទ្រាំដែលអាចទ្រទ្រង់យើងបានតាមរយៈពេលវេលាដ៏ខ្មៅងងឹត។

3. ស្វែងរកអត្ថន័យក្នុងការរងទុក្ខ:
បទពិសោធន៍របស់ហ្វ្រេនលិនក្នុងជំរុំប្រមូលផ្តុំណាស៊ីបានបង្ហាញពីគំនិតដែលបុគ្គលអាចរកឃើញអត្ថន័យក្នុងការរងទុក្ខយ៉ាងខ្លាំងបំផុត។ការរកឃើញនេះជំទាស់នឹងទស្សនៈដែលថាជីវិតដែលមានអត្ថន័យត្រូវតែមានការឈឺចាប់។

4. ឆន្ទៈដែលមានន័យថា:
Frankl ណែនាំគំនិតនៃឆន្ទៈក្នុងន័យថាជាកម្លាំងជំរុញក្នុងម្នាក់ៗ។នេះនឹងជួយឱ្យយើងទទួលបានកាលៈទេសៈពិបាកដោយស្វែងរកគោលបំណង។

5. ផ្លូវពិសេសមួយដែលមានន័យខ្លឹមសារ:
អត្ថន័យមិនមែនជាគំនិតទាំងអស់ដែលមានទំហំសមស្របទេ។ដំណើររបស់មនុស្សម្នាក់ៗដើម្បីស្វែងរកអត្ថន័យគឺមានតែមួយគត់និងស៊ីជម្រៅ។វាអាចពាក់ព័ន្ធនឹងទំនាក់ទំនងការងារការច្នៃប្រឌិតឬសូម្បីតែវិធីមួយឆ្លើយតបនឹងការរងទុក្ខ។

6. ត្រីកោណកោង:
ហ្វ្រេនល៍ពិពណ៌នាអំពីទ្រីយ៉ាកសោកនាដកម្ម - ការឈឺចាប់កំហុសនិងការស្លាប់ - ជាមូលដ្ឋានគ្រឹះនៃភាពជាមូលដ្ឋាននៃអត្ថិភាពរបស់មនុស្ស។ទោះយ៉ាងណាវាស្ថិតនៅក្នុងបរិបទនៃទ្រីយ៉ាឌីនេះដែលបុគ្គលម្នាក់ៗអាចរកឃើញអត្ថន័យយ៉ាងជ្រាលជ្រៅ។

ការទទួលខុសត្រូវក្នុងការស្វែងរកអត្ថន័យ:
ហ្វ្រែនល៍អះអាងថាយើងម្នាក់ៗមានទំនួលខុសត្រូវក្នុងការស្វែងរកអត្ថន័យនៅក្នុងជីវិតរបស់យើង។ការទទួលខុសត្រូវនេះមិនមែនជាបន្ទុកទេប៉ុន្តែជាឱកាសដើម្បីចូលរួមចំណែកដល់ពិភពលោកក្នុងរបៀបផ្ទាល់ខ្លួនរបស់យើង។

8. សេរីភាពក្នុងការស្វែងរកអត្ថន័យ:
ក្នុងការស្វែងរកអត្ថន័យបុគ្គលរកឃើញនូវសេរីភាពដ៏ពិសេសមួយ។សេរីភាពនេះមិនមែនសេរីភាពពីការទទួលខុសត្រូវឬការរងទុក្ខទេប៉ុន្តែសេរីភាពក្នុងការជ្រើសរើសឥរិយាបថរបស់មនុស្សម្នាក់និងការឆ្លើយតបចំពោះបញ្ហាប្រឈមក្នុងជីវិត។

9. ការខកចិត្តដែលមានស្រាប់:
Frankl ព្រមានពីផលវិបាកនៃការខកចិត្តដែលមាន - អារម្មណ៍នៃភាពទទេនិងភាពអស់សង្ឃឹមដែលអាចបណ្តាលមកពីការខកខានមិនបានស្វែងរកអត្ថន័យ។ការខកចិត្តនេះអាចនាំឱ្យមានទំរង់ផ្សេងៗនៃភាពតានតឹងខាងផ្លូវចិត្ត។
