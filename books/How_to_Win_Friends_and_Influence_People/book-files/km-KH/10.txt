ភាគទី 9 ៈប្រសិនបើអ្នកខុសសូមសារភាពវា

នៅក្នុងសៀវភៅនេះនៃសៀវភៅដ៏ជ្រាលជ្រៅរបស់វីកតូអ៊ីហ្វ្រេនអិនស្វែងរកអត្ថន័យយើងមានសារៈសំខាន់ក្នុងការទទួលស្គាល់កំហុសរបស់យើងឱបភាពរាបទាបនិងការស្វែងរកអត្ថន័យសូម្បីតែក្នុងកាលៈទេសៈពិបាកក៏ដោយ។

អំណាចនៃការបន្ទាបខ្លួន:
ហ្វ្រេនអិលអិលបង្ហាញពីសារៈសំខាន់នៃការបន្ទាបខ្លួនក្នុងអន្តរកម្មរបស់យើងជាមួយអ្នកដទៃនិងក្នុងការលូតលាស់ផ្ទាល់ខ្លួនរបស់យើង។សារភាពនៅពេលដែលយើងខុសឬនៅពេលដែលយើងបានធ្វើខុសគឺជាសញ្ញានៃការបន្ទាបខ្លួនដែលជំរុញឱ្យមានទំនាក់ទំនងសុខភាពនិងការអភិវឌ្ឍន៍ផ្ទាល់ខ្លួន។

រៀនពីកំហុស:
ហ្វ្រែនចលើកទឹកចិត្តអាមេរិកឱ្យចាត់ទុកកំហុសជាឱកាសសម្រាប់ការរៀនសូត្រនិងការរីកចម្រើន។នៅពេលយើងទទួលយកកំហុសរបស់យើងយើងបើកទ្វារឱ្យកែលម្អដោយខ្លួនឯងនិងការយល់ដឹងកាន់តែស៊ីជម្រៅអំពីខ្លួនយើង។

3. ភាពចម្លែកនៃការរងទុក្ខនិងអត្ថន័យ:
ប្រធានបទសំខាន់មួយនៅក្នុងការងាររបស់ហ្វ្រែនលគឺជាគំនិតដែលថាសូម្បីតែប្រឈមមុខនឹងការរងទុក្ខនិងភាពលំបាកក៏ដោយយើងអាចរកឃើញអត្ថន័យផងដែរ។ការទទួលស្គាល់កំហុសរបស់យើងនិងការឱបក្រសោបការបន្ទាបខ្លួនអាចជាផ្នែកមួយនៃដំណើរដ៏មានអត្ថន័យនោះ។

4. ការឆ្លុះបញ្ចាំងដោយខ្លួនឯងនិងការរកឃើញដោយខ្លួនឯង:
តាមរយៈការពិនិត្យមើលសកម្មភាពរបស់យើងដោយស្មោះត្រង់និងការទទួលស្គាល់នៅពេលដែលយើងខុសយើងចូលរួមក្នុងដំណើរការនៃការឆ្លុះបញ្ចាំងដោយខ្លួនឯងនិងការរកឃើញដោយខ្លួនឯង។ការប្រកាន់យកនេះអនុញ្ញាតឱ្យយើងតម្រឹមសកម្មភាពរបស់យើងជាមួយនឹងគុណតម្លៃនិងគោលការណ៍របស់យើង។

តួនាទីនៃគណនេយ្យភាព:
គណនេយ្យភាពសម្រាប់កំហុសរបស់យើងគឺចាំបាច់មិនត្រឹមតែនៅក្នុងទំនាក់ទំនងផ្ទាល់ខ្លួនប៉ុណ្ណោះទេប៉ុន្តែថែមទាំងនៅក្នុងសង្គមផងដែរ។លោកហ្វ្រែនលិមបានណែនាំថាការទទួលខុសត្រូវចំពោះសកម្មភាពរបស់យើងរួមចំណែកដល់ភាពប្រសើរនៃមនុស្សជាតិទាំងមូល។

6. ភាពត្រឹមត្រូវនិងភាពស្មោះត្រង់:
ទទួលស្គាល់នៅពេលដែលយើងខុសគឺជាការបង្ហាញភាពត្រឹមត្រូវនិងភាពស្មោះត្រង់។វាឆ្លុះបញ្ចាំងពីការប្តេជ្ញាចិត្តរបស់យើងក្នុងការរស់នៅស្របតាមគុណតម្លៃនិងគោលការណ៍របស់យើង។

7. ការព្យាបាលនិងការផ្សះផ្សា:
ការទទួលស្គាល់កំហុសរបស់យើងអាចនាំឱ្យមានការព្យាបាលនិងការផ្សះផ្សាក្នុងទំនាក់ទំនងដែលអាចមានភាពតានតឹងដោយការយល់ច្រឡំឬជម្លោះ។

8. ការរំដោះការសុំអភ័យទោស:
សុំទោសនិងការទទួលស្គាល់កំហុសរបស់យើងអាចត្រូវបានរំដោះ។វាជួយកាត់បន្ថយការទទួលបន្ទុកនៃកំហុសហើយអនុញ្ញាតឱ្យយើងឆ្ពោះទៅមុខជាមួយនឹងមនសិការកាន់តែច្បាស់។

9. ភាពក្លាហានក្នុងការប្រឈមមុខនឹងការពិត:
ទទួលស្គាល់នៅពេលដែលយើងខុសត្រូវការភាពក្លាហានព្រោះវាពាក់ព័ន្ធនឹងការប្រឈមមុខនឹងភាពពិតនៃសកម្មភាពនិងផលវិបាករបស់ពួកគេ។ហ្វ្រេនអិលលើកទឹកចិត្តឱ្យយើងកោះហៅភាពក្លាហាននេះដែលជាផ្នែកមួយនៃដំណើររបស់យើងឆ្ពោះទៅរកការរកឃើញដោយខ្លួនឯងនិងអត្ថន័យ។
